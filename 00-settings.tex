\defbeamertemplate{footline}{customfootline}{
%	\vskip-20pt
	\leavevmode%
	\color{lightgray}\rule{\paperwidth}{0.5pt}
	\hbox{%
		\begin{beamercolorbox}[wd=.45\paperwidth,ht=8ex,dp=1ex,left]{myfootline}%
			\hspace{0.5cm}
			\vspace{0.03cm}
			\includegraphics[height=0.5cm,keepaspectratio]{\getFootlineLogo} % adjust size accordingly
		\end{beamercolorbox}%
		\begin{beamercolorbox}[wd=.1\paperwidth,ht=8ex,dp=1ex,center]{myfootline}%
			\vspace{0.2cm}
			\usebeamerfont{title in head/foot}\insertframenumber
		\end{beamercolorbox}%
		\begin{beamercolorbox}[wd=.45\paperwidth,ht=8ex,dp=1ex,right]{myfootline}%
			\vspace{0.2cm}
			%    \usebeamerfont{author in head/foot}
			\getLocationName,~\insertdate,~\getStudentName \hspace{0.5cm}
	\end{beamercolorbox}}%
	\vskip0pt%
}
